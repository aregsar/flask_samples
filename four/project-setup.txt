#create the project directory and cd into it
$ mkdir four
$ cd four

# create virtual environment and venv subfolder (one time operation)
$ virtualenv -p $(which python) venv

# activate the virtual environment
$ . venv/bin/activate

#install the flask package
(venv)$ pip install Flask==0.10.1

#create requirements.txt file populating it with current installed packages
(venv)$ pip freeze > requirements.txt

#install or reinstall python packages listed in requirements file into venv
(venv)$ pip install -r requirements.txt

#create application file
(venv)$ touch app.py

#run the appication
(venv)$ python app.py

#de activate the virtual environment
(venv)$ deactivate

# re-activate the virtual environment
$ . venv/bin/activate
